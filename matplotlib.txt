regimen_group = data_no_duplicates.groupby('Drug Regimen')['Tumor Volume (mm3)']

regimen_stats_df = pd.DataFrame(regimen_group)
regimen_stats_df ['Mean'] = data_no_duplicates['Tumor Volume (mm3)'].mean()
regimen_stats_df ['Median'] = data_no_duplicates['Tumor Volume (mm3)'].median()
regimen_stats_df ['Variance'] = data_no_duplicates['Tumor Volume (mm3)'].var()
regimen_stats_df ['Sandard Deviation'] = data_no_duplicates['Tumor Volume (mm3)'].std()
regimen_stats_df ['SEM'] = data_no_duplicates['Tumor Volume (mm3)'].sem()
